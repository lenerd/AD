\begin{algorithm}
    \caption{\textsc{k-highest Comparison}}
    \begin{algorithmic}[1]
        \Procedure{Check-Node}{heap, index, x, k, count}
        \If {$node \geq k$}
            \State $current \gets current + 1$
            \If {$count \geq k$}
                \State \Return $count$
            \EndIf
            \State $count \gets \Call{Check-Node}{heap, 2 \cdot index, k, x, count}$
            \If {$count \geq k$}
                \State \Return $count$
            \EndIf
            \State \Return \Call{Check-Node}{$heap, 2 \cdot index + 1, k, x, count$}
        \EndIf
        \EndProcedure
        \Procedure{k-highest Comparison}{H, k, x}
        \If {$\Call{Check-Node}{H, 1, k, x, 0} \geq k$}
            \State \Return true
        \Else
            \State \Return false
        \EndIf
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
